Layout Calculator:

ts
// utils/layoutCalculator.ts
export const calculateGrid = (containerWidth: number) => {
  const sidebarWidth = 256 + 48; // Left + right
  const availableWidth = containerWidth - sidebarWidth - 32; // Padding
  return Math.floor(availableWidth / 248); // Returns optimal column count
};
3. Performance Enhancements
Dynamic Column Adjustment:

tsx
const [columnCount, setColumnCount] = useState(4);

useEffect(() => {
  const handleResize = () => {
    const container = document.querySelector('.product-grid-container');
    if (container) {
      setColumnCount(calculateGrid(container.clientWidth));
    }
  };
  
  window.addEventListener('resize', handleResize);
  return () => window.removeEventListener('resize', handleResize);
}, []);
