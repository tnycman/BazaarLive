// EnterpriseProductGrid.tsx
const EnterpriseProductGrid = ({ targetWidth, sidebarOffset }) => {
  const containerRef = useRef(null);
  const [columnCount, setColumnCount] = useState(4);

  // Debounced resize handler
  useLayoutEffect(() => {
    const calculateColumns = () => {
      if (containerRef.current) {
        const availableWidth = containerRef.current.clientWidth - sidebarOffset;
        setColumnCount(Math.floor(availableWidth / targetWidth));
      }
    };

    const debouncedCalc = debounce(calculateColumns, 100);
    window.addEventListener('resize', debouncedCalc);
    calculateColumns(); // Initial calc
    return () => window.removeEventListener('resize', debouncedCalc);
  }, [targetWidth, sidebarOffset]);

  return (
    <div 
      ref={containerRef}
      className={`grid gap-4 p-4`}
      style={{
        gridTemplateColumns: `repeat(${columnCount}, minmax(${targetWidth}px, 1fr))`
      }}
    >
      {products.map(product => (
        <ProductCard key={product.id} />
      ))}
    </div>
  );
};

// utils/debounce.ts
export const debounce = (fn: Function, delay: number) => {
  let timeoutId: NodeJS.Timeout;
  return (...args: any[]) => {
    clearTimeout(timeoutId);
    timeoutId = setTimeout(() => fn(...args), delay);
  };
};
